<!DOCTYPE html>

<html>

<head>

    <title>Music Genre Classification</title>

    <script src="https://code.jquery.com/jquery-3.6.0.min.js"></script>

    <script>

        $(document).ready(function() {

            // Function to handle the form submission

            $("#predictionForm").submit(function(event) {

                event.preventDefault();

                var formData = new FormData($(this)[0]);

                $.ajax({

                    url: '/predict_genre',

                    type: 'POST',

                    data: formData,

                    processData: false,

                    contentType: false,

                    success: function(response) {

                        var predictedGenre = response.predicted_genre;

                        $("#result").text("Predicted Genre: " + predictedGenre);

                    }

                });

            });

        });

    </script>

</head>

<body>

    <h1>Music Genre Classification</h1>

    <form id="predictionForm">

        <label for="audioFile">Select an audio file:</label>

        <input type="file" id="audioFile" name="audioFile" accept="audio/*" required>

        <button type="submit">Predict Genre</button>

    </form>

    <div id="result"></div>

</body>

</html>
